2.2 Weighted Score Calculation
javascript
function calculateVPNScore(v) {
  const weights = {
    speed: 0.20,      // 0-100
    privacy: 0.25,    // 0-100 (audit + no-logs + jurisdiction)
    security: 0.15,   // 0-100
    streaming: 0.10,  // 0-100 (how many platforms)
    value: 0.10,      // 0-100
    ethics: 0.10,     // 0-100 (transparency + ownership)
    apps: 0.05,       // 0-100 (how many platforms)
    connections: 0.05 // 0-100 (unlimited = 100, 10 = 80, 5 = 60)
  };
  
  return Math.round(
    v.speedScore * weights.speed +
    v.privacyScore * weights.privacy +
    v.securityScore * weights.security +
    v.streamingScore * weights.streaming +
    v.valueScore * weights.value +
    v.ethicsScore * weights.ethics +
    v.appsScore * weights.apps +
    v.connectionsScore * weights.connections
  );
}
2.3 Privacy Score Calculator
javascript
function calculatePrivacyScore(vpn) {
  let score = 0;
  
  // No-logs audit (max 40 points)
  if (vpn.auditType === 'independent') score += 40;
  else if (vpn.auditType === 'self-published') score += 20;
  else if (vpn.auditType === 'none') score += 0;
  
  // Jurisdiction (max 30 points)
  const safeJurisdictions = ['Panama', 'Switzerland', 'Iceland', 'Malaysia', 'British Virgin Islands', 'Gibraltar'];
  const mediumJurisdictions = ['Canada', 'Germany', 'Netherlands', 'Spain'];
  if (safeJurisdictions.includes(vpn.jurisdiction)) score += 30;
  else if (mediumJurisdictions.includes(vpn.jurisdiction)) score += 15;
  else score += 5;
  
  // Warrant canary (10 points)
  if (vpn.warrantCanary) score += 10;
  
  // RAM-only servers (10 points)
  if (vpn.ramServers) score += 10;
  
  // Anonymous payment options (10 points)
  if (vpn.acceptsCrypto) score += 5;
  if (vpn.acceptsCash) score += 5;
  
  return score;
}
2.4 Ethics & Ownership Score
javascript
function calculateEthicsScore(vpn) {
  let score = 0;
  
  // KAPE framework (Transparency)
  // K = Known ownership (10 pts)
  if (vpn.parentCompany !== 'Unknown') score += 10;
  else score += 0;
  
  // A = Audits published (20 pts)
  if (vpn.auditType === 'independent') score += 20;
  else if (vpn.auditType === 'self-published') score += 10;
  
  // P = Past incidents (20 pts for none, -20 for breach)
  if (vpn.pastBreaches === 'none') score += 20;
  else if (vpn.pastBreaches === 'resolved') score += 5;
  else if (vpn.pastBreaches === 'unresolved') score += -20;
  
  // E = Ethical leadership (20 pts)
  if (vpn.founderBackground === 'transparent') score += 20;
  else if (vpn.founderBackground === 'unknown') score += 5;
  
  // Bonus: Open source (20 pts)
  if (vpn.openSource) score += 20;
  
  // Bonus: Non-profit or privacy foundation (10 pts)
  if (vpn.nonProfit) score += 10;
  
  return Math.min(100, score);
}
2.5 Speed Score (vs advertised)
javascript
// Speed is measured as % of advertised bandwidth
// Example: Advertised 100 Mbps, real 85 Mbps = 85% = 85 points
function calculateSpeedScore(vpn) {
  // Real-world speed as % of advertised
  const percentOfAdvertised = (vpn.realSpeedMbps / vpn.advertisedSpeedMbps) * 100;
  return Math.min(100, percentOfAdvertised);
}
